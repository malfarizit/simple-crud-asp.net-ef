//Pivot Table query
SELECT * FROM (
  SELECT
    [id],
    [Type],
	[Gender],
    [Number_of_cats]
  FROM numberofcats
) Results
PIVOT (
  SUM([Number_of_cats])
  FOR [Type]
  IN (
    [Persian],
    [Angora],
	[Sphynx],
	[Mainecoon]
  )
) AS PivotTable
